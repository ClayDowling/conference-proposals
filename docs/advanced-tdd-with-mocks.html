The Problem

class CartItem {
	private:
		string name;
		int quantity;
		double price;
	public:
		const string &getName() const;
		void setName(const string &name);
		int getQuantity() const;
		void setQuantity(int quantity);
		double getPrice() const;
		void setPrice(double price);

		double getItemTotal();
};

The Test


TEST_F(CITest, getLineTotal_withQuantityOne_returnsItemPrice)
{
	item.setName(ITEM_NAME);
	item.setPrice(3.14);
	item.setQuantity(1);

	ASSERT_EQ(item.getItemTotal(), 3.14);
}

Minimum Effort


double CartItem::getItemTotal() {
	return 3.14;
}

Fail More!


TEST_F(CITest, getLineTotal_withDifferentPrice_returnsItemPrice)
{
	item.setName(ITEM_NAME);
	item.setPrice(2.71);
	item.setQuantity(2);

	ASSERT_EQ(item.getItemTotal(), 5.42);
}

100% Test Coverage!


double CartItem::getItemTotal() {
